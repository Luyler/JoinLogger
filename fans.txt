local player = game.Players.LocalPlayer

function webhook(webhook,str)
    local Response = syn.request({
        Url = webhook;
        Method = "POST";
        Headers = {
            ["Content-Type"] = "application/json";
        };
        Body = game:GetService("HttpService"):JSONEncode({["content"] = str})
    })    
end
